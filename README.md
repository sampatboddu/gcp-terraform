# create terraform-admin service account

# enable cloud resource API, Cloud Storage & Compute Engine API's

# create main.tf file with provider resource and terraform backend resource

provider "google" {
  version = "3.5.0"
  credentials = file("<path-to-service-account-key-file>.json")
  project = "<project-id>"
  region  = "us-central1"
  zone    = "us-central1-c"
}

# terraform state
terraform {
  backend "gcs" {
    bucket = "<bucket>"
    prefix = "terraform/state"
    
   }
}

# terraform init initializes your current working directory that contains your configuration files
terraform init

# terraform plan looks within your current directory for config files and state files to refresh. 
# It allows you to see if your execution plan matches your expectations   
terraform plan

# terraform apply to apply changes to your infrastructure that is generated by the execution plan
terraform apply

# terraform destroy to destroy the resources in your infrastructure
terraform destroy

# module

  source  = "terraform-google-modules/network/google"

      subnet_name   = "subnet-01"
      subnet_ip     = var.cidrs
      subnet_region = var.region

    },
  ]

    subnet-01 = []

  }
}
